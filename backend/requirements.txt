# Default OCR engine: a thin wrapper around the system `tesseract` binary
# (app/ai/ocr.py -> TesseractOcrEngine). No native ML framework wheel to
# chase, so it installs identically on every Python version/platform,
# including Python 3.13 on Apple Silicon. Requires the `tesseract` binary
# on PATH:
#   macOS:  brew install tesseract
#   Debian/Ubuntu: apt-get install tesseract-ocr
# Imported lazily and optional at runtime — see app/ai/ocr.py. If the
# binary isn't installed, the API still starts and `/registration/analyze`
# still works (in manual-entry mode).
pytesseract==0.3.13

# Alternative OCR engine (app/ai/ocr.py -> PaddleOcrEngine), NOT installed
# by default. Uncomment to opt in — confirmed to ship Python 3.13 + macOS
# arm64 (Apple Silicon) wheels as of mid-2026, but is a much heavier native
# dependency, and its 3.x Python API is a breaking change from the 2.x
# calling convention `PaddleOcrEngine.extract()` currently uses (see that
# class's docstring before switching `get_ocr_engine()` over to it).
# paddlepaddle==3.3.1
# paddleocr==3.7.0

# Default barcode/QR engine: zxing-cpp (app/ai/barcode.py -> ZxingBarcodeEngine).
# Pure C++20 decoder statically bundled inside the wheel — no external
# shared library to locate at runtime, unlike pyzbar/zbar (see below).
# Ships Python 3.13 + macOS arm64 (Apple Silicon) wheels. Optional at
# runtime — see app/ai/barcode.py; if missing, the API still starts and
# registration falls back to OCR automatically.
zxing-cpp==3.1.0

# Alternative barcode engine (app/ai/barcode.py -> PyzbarBarcodeEngine),
# NOT installed by default. Requires the system `zbar` shared library —
#   macOS:  brew install zbar
#   Debian/Ubuntu: apt-get install libzbar0
# — which has a long history of "Unable to find zbar shared library"
# failures on Apple Silicon (Homebrew's non-standard install path, and/or
# an architecture mismatch between the Python interpreter and the
# installed dylib) even when the library is genuinely installed. That
# failure class is exactly why zxing-cpp is the default above; uncomment
# this only if you already have a working zbar install and prefer it.
# pyzbar==0.1.9
